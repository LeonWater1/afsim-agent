platform_type acoustic_target WSF_PLATFORM
   icon aircraft
   mover WSF_AIR_MOVER
      maximum_speed 200 m/sec
      minimum_speed 50 m/sec
      default_radial_acceleration 3.0 g
   end_mover
   acoustic_signature TARGET_SIG
end_platform_type

acoustic_signature TARGET_SIG
   constant 70 dB
end_acoustic_signature

antenna_pattern ACOUSTIC_ANT
   constant_pattern
      peak_gain 0 db
      azimuth_beamwidth 360 deg
      elevation_beamwidth 360 deg
   end_constant_pattern
end_antenna_pattern

sensor ACOUSTIC_SENSOR WSF_ESM_SENSOR
   selection_mode multiple

   mode_template
      receiver
         antenna_pattern ACOUSTIC_ANT
         noise_power -160 dBw
         bandwidth 100.0 khz
      end_receiver
   end_mode_template

   mode LISTEN
      scan_mode fixed
      azimuth_scan_limits 0 deg 0 deg
      elevation_scan_limits 0 deg 0 deg
   end_mode
end_sensor

platform_type sensor_station WSF_PLATFORM
   icon ground_sensor
   mover WSF_GROUND_MOVER end_mover
   sensor acoustic ACOUSTIC_SENSOR
   end_sensor
   processor status WSF_SCRIPT_PROCESSOR
      update_interval 5.0 sec
      on_initialize
         print("Acoustic sensor station activated.")
      end_on_initialize
      on_update
         print("Time: ", TIME_NOW, " sec")
      end_on_update
   end_processor
end_platform_type

platform moving_target acoustic_target
   side blue
   position 35.0n 120.0w altitude 5000 ft msl
   route
      position 35.0n 120.0w altitude 5000 ft msl speed 100 m/sec
      position 36.0n 120.5w altitude 5000 ft msl speed 100 m/sec
      position 36.5n 121.0w altitude 5000 ft msl speed 100 m/sec
   end_route
end_platform

platform sensor_post sensor_station
   side red
   position 35.2n 120.1w altitude 0 ft agl
   processor status on end_processor
end_platform

event_pipe
   file output/acoustic_demo.aer
end_event_pipe

end_time 600 sec
end_platform
end_platform_type
