script_interface
   debug
end_script_interface

platform_type MY_STRIKER WSF_PLATFORM
   side blue
   mover WSF_AIR_MOVER
      maximum_speed 0.000000001 knots
      minimum_speed 0 knots
      maximum_altitude 50000 ft
      minimum_altitude 0 ft
      at_end_of_path stop
   end_mover
   processor mover_script WSF_SCRIPT_PROCESSOR
      update_interval 0.1 sec
      script_variables
         int routeIndex = 0;
         double speedMPS = 200.0;
      end_script_variables
      on_update
         WsfRoute rowt = PLATFORM.Route();
         if (!rowt.IsValid()) {
            return;
         }
         if (routeIndex >= rowt.Size()) {
            return;
         }
         double delta = PROCESSOR.UpdateInterval();
         double myDist = delta * speedMPS;
         WsfGeoPoint a = PLATFORM.Location();
         WsfGeoPoint b = rowt.Get(routeIndex).Location();
         double dist = a.SlantRangeTo(b);
         Vec3 direction = Vec3.Subtract(b.LocationWCS(), a.LocationWCS());
         if (myDist >= dist) {
            routeIndex += 1;
            if (routeIndex >= rowt.Size()) {
               PLATFORM.SetLocation(b);
               return;
            }
            double remaining = myDist - dist;
            WsfGeoPoint c = rowt.Get(routeIndex).Location();
            direction = Vec3.Subtract(c.LocationWCS(), b.LocationWCS());
            if (direction.Magnitude() > 0.0) {
               direction.Normalize();
               direction.Scale(remaining);
               PLATFORM.SetLocation(WsfGeoPoint.FromLocationWCS(b.LocationWCS() + direction));
            }
         } else {
            if (direction.Magnitude() > 0.0) {
               direction.Normalize();
               direction.Scale(myDist);
               PLATFORM.SetLocation(WsfGeoPoint.FromLocationWCS(a.LocationWCS() + direction));
            }
         }
      end_on_update
   end_processor
end_platform_type
