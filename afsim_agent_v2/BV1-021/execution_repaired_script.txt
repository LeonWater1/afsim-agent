script_interface
   debug
end_script_interface

platform_type MY_STRIKER WSF_PLATFORM
   side blue
   mover WSF_AIR_MOVER
      maximum_speed 0.000000001 knots
      minimum_speed 0 knots
      maximum_altitude 50000 ft
      minimum_altitude 0 ft
      at_end_of_path stop
   end_mover
   processor mover_script WSF_SCRIPT_PROCESSOR
      update_interval 0.1 sec
      script_variables
         int routeIndex = 0;
         double speedMPS = 200.0;
      end_script_variables
      on_update
         WsfRoute rowt = PLATFORM.Route();
         if (!rowt.IsValid()) {
            return;
         }
         if (routeIndex >= rowt.Size()) {
            return;
         }
         double delta = PROCESSOR.UpdateInterval();
         double myDist = delta * speedMPS;
         WsfGeoPoint a = PLATFORM.Location();
         WsfGeoPoint b = rowt.Get(routeIndex).Location();
         double dist = a.SlantRangeTo(b);
         double heading = a.HeadingTo(b);
         if (myDist >= dist) {
            routeIndex = routeIndex + 1;
            if (routeIndex >= rowt.Size()) {
               PLATFORM.SetLocation(b);
               return;
            }
            double remaining = myDist - dist;
            WsfGeoPoint c = rowt.Get(routeIndex).Location();
            heading = b.HeadingTo(c);
            WsfGeoPoint newPos = b.Offset(heading, remaining);
            PLATFORM.SetLocation(newPos);
         } else {
            WsfGeoPoint newPos = a.Offset(heading, myDist);
            PLATFORM.SetLocation(newPos);
         }
      end_on_update
   end_processor
end_platform_type

platform mover_demo_platform MY_STRIKER
   side blue
   route
      position 30.67n 104.07e altitude 3000 m msl speed 100 m/sec
      position 30.72n 104.12e altitude 3000 m msl speed 100 m/sec
   end_route
end_platform

event_pipe

end_event_pipe
