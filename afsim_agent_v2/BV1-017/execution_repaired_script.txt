   debug

comm BASIC_COMM WSF_COMM_TRANSCEIVER
end_comm

processor BASIC_TRACK_PROC WSF_TRACK_PROCESSOR
   report_interval 2 sec
   circular_report_rejection on
   update_interval 10 sec
   on_message
      type WSF_TRACK_MESSAGE
      script
         WsfTrackMessage tmsg = (WsfTrackMessage)MESSAGE;
         if(tmsg.Originator() != PLATFORM.Name())
         {
            print("T=", TIME_NOW, " P=", PLATFORM.Name(), " received track from ", tmsg.Originator());
         }
      end_script
   end_on_message
end_processor

group GROUP1 WSF_GROUP end_group
group GROUP2 WSF_GROUP end_group

platform_type BASIC_STATION WSF_PLATFORM
   mover WSF_GROUND_MOVER
      maximum_speed 1 m/sec
      minimum_speed 0 m/sec
   end_mover

   comm PLATFORM_COMM WSF_COMM_TRANSCEIVER
      internal_link TRACK_PROC
   end_comm

   processor TRACK_PROC WSF_TRACK_PROCESSOR
      report_interval 2 sec
      circular_report_rejection on
      update_interval 10 sec
      on_message
         type WSF_TRACK_MESSAGE
         script
            WsfTrackMessage tmsg = (WsfTrackMessage)MESSAGE;
            if(tmsg.Originator() != PLATFORM.Name())
            {
               print("T=", TIME_NOW, " P=", PLATFORM.Name(), " received track from ", tmsg.Originator());
            }
         end_script
      end_on_message
   end_processor

   sensor radar WSF_RADAR_SENSOR
      frame_time 2 sec
      reports_location
      transmitter
         power 100 kw
         frequency 1000 mhz
         antenna_pattern
            constant_pattern
               peak_gain 30 db
               azimuth_beamwidth 10 deg
               elevation_beamwidth 10 deg
            end_constant_pattern
         end_antenna_pattern
      end_transmitter
      receiver
         antenna_pattern
            constant_pattern
               peak_gain 30 db
               azimuth_beamwidth 10 deg
               elevation_beamwidth 10 deg
            end_constant_pattern
         end_antenna_pattern
      end_receiver
   end_sensor
end_platform_type
