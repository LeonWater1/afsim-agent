# UNCLASSIFIED
#
# Description: Group communication scenario.
#
# Scenario:
#  - Platforms communicate track data within and across groups
#

# Shared Comm and Processor Blocks
comm datalink_basic WSF_COMM_TRANSCEIVER
end_comm

comm group_comm_basic WSF_COMM_TRANSCEIVER
end_comm

processor track_processor_basic WSF_TRACK_PROCESSOR
end_processor

# Platform Type Declarations
platform_type PT_DETECTOR_UNIT WSF_PLATFORM
   comm group_network datalink_basic
      internal_link report_processor
   end_comm
   processor report_processor track_processor_basic
   end_processor
   sensor geo WSF_GEOMETRIC_SENSOR
      off
      ignore_same_side
      reports_location
      maximum_range 100 nm
      frame_time 1 sec
      internal_link report_processor
   end_sensor
end_platform_type

platform_type PT_RELAY_UNIT WSF_PLATFORM
   comm group_network group_comm_basic
      internal_link track_proc
   end_comm
   processor track_proc WSF_TRACK_PROCESSOR
   end_processor
end_platform_type

platform_type PT_OBSERVER_UNIT WSF_PLATFORM
   comm group_network group_comm_basic
   end_comm
end_platform_type

# Groups
group BROADCAST_DET WSF_GROUP end_group
group BROADCAST_OBS WSF_GROUP end_group

# Platform Instances
platform detector_unit PT_DETECTOR_UNIT
   side blue
   position 30.67n 104.07e
   group BROADCAST_DET
   edit comm group_network group_join BROADCAST_DET
   edit processor report_processor report_to_group BROADCAST_DET via group_network
   edit sensor geo on
end_platform

platform relay_unit PT_RELAY_UNIT
   side blue
   position 30.70n 104.10e
   group BROADCAST_DET
   group BROADCAST_OBS
   edit comm group_network group_join BROADCAST_DET
   edit comm group_network group_join BROADCAST_OBS
   edit processor track_proc report_to_group BROADCAST_OBS via group_network
end_platform

platform observer_unit PT_OBSERVER_UNIT
   side blue
   position 30.73n 104.13e
   group BROADCAST_OBS
   edit comm group_network group_join BROADCAST_OBS
end_platform

# Event handling
event_pipe
end_event_pipe

event_output
   file "output.events"
end_event_output

end_time 240 sec
