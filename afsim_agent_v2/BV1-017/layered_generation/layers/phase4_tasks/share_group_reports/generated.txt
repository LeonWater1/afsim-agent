group broadcast_template WSF_GROUP end_group

script void CreateBroadcastGroups()
   WsfSimulation.CreateGroup("BROADCAST_DET","broadcast_template");
   WsfSimulation.CreateGroup("BROADCAST_OBS","broadcast_template");
end_script

observer
   enable SIMULATION_INITIALIZING CreateBroadcastGroups
end_observer

platform detector_unit PT_DETECTOR_UNIT
   edit comm datalink group_join BROADCAST_DET end_comm
   edit processor track_proc report_to_group BROADCAST_DET via datalink end_processor
   edit sensor geo on end_sensor
end_platform

platform relay_unit PT_RELAY_UNIT
   edit comm datalink group_join BROADCAST_DET end_comm
   edit comm datalink group_join BROADCAST_OBS end_comm
   edit processor track_proc report_to_group BROADCAST_OBS via datalink end_processor
end_platform

platform observer_unit PT_OBSERVER_UNIT
   edit comm datalink group_join BROADCAST_OBS end_comm
end_platform
