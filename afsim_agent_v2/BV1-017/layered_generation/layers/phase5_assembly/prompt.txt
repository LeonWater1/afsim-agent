# ****************************************************************************
# CUI
#
# The Advanced Framework for Simulation, Integration, and Modeling (AFSIM)
#
# The use, dissemination or disclosure of data in this file is subject to
# limitation or restriction. See accompanying README and LICENSE for details.
# ****************************************************************************

# UNCLASSIFIED
#
# Description: This simple scenario provides an example of transmitting 
#   tracks received from off-board platforms to other groups.
#
# Scenario:
#  - Platform 1 (p1) is the only platform that has their sensor ON in the simulation
#  - Platform 1 (p1) can communicate tracks to platforms in FIGHTER_GROUP_1 (p1, p2)
#  - Platform 2 (p2) can communicate tracks to platforms in FIGHTER_GROUP_2 (p2, p3)
#  - p1 CANNOT communicate tracks to p3, but must go by way of p2
#

# -- Basic Subsystems --
comm BASIC_COMM WSF_COMM_TRANSCEIVER 
   #debug 
end_comm

processor BASIC_TRACK_PROC WSF_TRACK_PROCESSOR
   report_interval 2 sec
   circular_report_rejection on
   update_interval 10 sec

   on_message
      type WSF_TRACK_MESSAGE
      script
         WsfTrackMessage tmsg = (WsfTrackMessage)MESSAGE;
         if(tmsg.Originator() != PLATFORM.Name())
         {
            print("T=",TIME_NOW," P=",PLATFORM.Name(),"  Track ",tmsg.Track().TargetName()," received from ",tmsg.Originator());
         }
      end_script
   end_on_message
end_processor

# -- Groups --
group fg_template WSF_GROUP end_group

script void CreateGroups()
   WsfSimulation.CreateGroup("FIGHTER_GROUP_1","fg_template");
   WsfSimulation.CreateGroup("FIGHTER_GROUP_2","fg_template");
end_script

observer
   enable SIMULATION_INITIALIZING CreateGroups
end_observer

# -- Platform Type --
platform_type BASIC_FIGHTER WSF_PLATFORM
   icon fighter
   mover WSF_AIR_MOVER end_mover

   comm datalink BASIC_COMM
      internal_link track_proc
   end_comm

   processor track_proc BASIC_TRACK_PROC
      script bool is_track_reportable(WsfTrack aTrack)
         writeln("T=",TIME_NOW," P=",PLATFORM.Name()," checking if ",aTrack.TargetName()," is reportable: YES");
         return true;
      end_script

      script void on_update()
      writeln(PLATFORM.Name());
      for (int i = 0; i < PLATFORM.MasterTrackList().TrackCount(); i = i + 1)
      {
         writeln(" * ", PLATFORM.MasterTrackList().TrackEntry(i).TrackId().ToString());
      }
      end_script
   end_processor

   sensor geo WSF_GEOMETRIC_SENSOR
      off
      ignore_same_side
      reports_location
      maximum_range 100 nm
      frame_time 1 sec
      internal_link track_proc
   end_sensor
end_platform_type

# ======================================================================================

# -- Blue Platforms ---
platform cmdr WSF_PLATFORM
   side blue
   position 1n 0e
end_platform

platform p1 BASIC_FIGHTER
   side blue
   #commander cmdr
   #group_join FIGHTER_GROUP_1

   edit comm datalink group_join FIGHTER_GROUP_1 end_comm

   route
      position 0n 01w altitude 10 kft heading 90 deg speed 700 kts
   end_route
   
   edit sensor geo on end_sensor
   
   edit processor track_proc
      report_to_group FIGHTER_GROUP_1 via datalink
      #report_to platform p3 comm datalink via datalink
   end_processor
end_platform

platform p2 BASIC_FIGHTER
   side blue
   #commander cmdr
   #group_join FIGHTER_GROUP_1
