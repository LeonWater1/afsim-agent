platform_type PT_FIRES_COMMANDER WSF_PLATFORM
  processor commander_task_proc WSF_TASK_PROCESSOR
    script void FireMission()
       WsfPlatform target = WsfSimulation.FindPlatform("target_unit");
       if (!target.IsValid())
          return;
       endif
       WsfTrack targetTrack = target.MakeTrack();
       targetTrack.SetAuxData("IMPACT_TIME", 300.0);

       WsfPlatform launcher;
       string weaponName;
       WsfWeaponTask task;

       launcher = WsfSimulation.FindPlatform("artillery_launcher");
       if (launcher.IsValid())
          weaponName = launcher.WeaponEntry(0).Name();
          task = WsfWeaponTask.Create("TIME_ON_TARGET", weaponName);
          task.SetQuantity(1);
          task.SetCommDevice("fires_comm");
          task.SetAssignee("artillery_launcher");
          AssignTask(targetTrack, task);
       endif

       launcher = WsfSimulation.FindPlatform("mortar_launcher");
       if (launcher.IsValid())
          weaponName = launcher.WeaponEntry(0).Name();
          task = WsfWeaponTask.Create("TIME_ON_TARGET", weaponName);
          task.SetQuantity(1);
          task.SetCommDevice("fires_comm");
          task.SetAssignee("mortar_launcher");
          AssignTask(targetTrack, task);
       endif

       launcher = WsfSimulation.FindPlatform("rocket_launcher");
       if (launcher.IsValid())
          weaponName = launcher.WeaponEntry(0).Name();
          task = WsfWeaponTask.Create("TIME_ON_TARGET", weaponName);
          task.SetQuantity(1);
          task.SetCommDevice("fires_comm");
          task.SetAssignee("rocket_launcher");
          AssignTask(targetTrack, task);
       endif
    end_script

    execute at_time 270 sec absolute
       FireMission();
    end_execute
  end_processor

  comm fires_comm WSF_COMM_TRANSCEIVER
  end_comm
end_platform_type
